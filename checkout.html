// ข้อมูลจำลองในตะกร้า (ปกติจะดึงมาจาก localStorage) const cart = [ { name: "Pain de Gêne (Almond Paste Sponge)", price: 450, weight: 0.5, type: 'normal' }, { name: "Peach Black Tea Entremet", price: 1200, weight: 1.0, type: 'chilled' } ]; function updateSummary() { const isBKK = document.querySelector('input[name="location"]:checked').value === 'BKK'; let subtotal = 0; let hasChilled = false; cart.forEach(item => { subtotal += item.price; if (item.type === 'chilled') hasChilled = true; }); // กำหนดค่าส่ง (ตัวอย่างเรทราคา) let shippingFee = 0; if (hasChilled) { shippingFee = isBKK ? 150 : 250; // รถเย็น document.getElementById('shipping-type').innerText = "❄️ รถควบคุมอุณหภูมิ"; } else { shippingFee = isBKK ? 50 : 80; // พัสดุธรรมดา document.getElementById('shipping-type').innerText = "📦 พัสดุธรรมดา"; } document.getElementById('subtotal').innerText = subtotal.toLocaleString(); document.getElementById('shipping-fee').innerText = shippingFee.toLocaleString(); document.getElementById('total-amount').innerText = (subtotal + shippingFee).toLocaleString(); } // เรียกให้คำนวณครั้งแรกตอนโหลดหน้า updateSummary();
